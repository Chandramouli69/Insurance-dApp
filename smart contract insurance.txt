pragma solidity ^0.8.30;

contract SimpleDecentralizedInsurance {
    enum ClaimStatus {
        NotFiled,
        Filed,
        Approved,
        Rejected,
        Paid
    }

    struct Claim {
        uint id;
        address payable customer;
        uint amount;
        ClaimStatus status;
        string description;
    }

    address public insurer;
    uint public claimCounter;

    mapping(uint => Claim) public claims;

    event ClaimFiled(
        uint id,
        address indexed customer,
        uint amount,
        string description
    );

    event ClaimApproved(uint id);
    event ClaimRejected(uint id);
    event ClaimPaid(uint id, uint amount);

    modifier onlyInsurer() {
        require(
            msg.sender == insurer,
            "Only insurer can perform this action"
        );
        _;
    }

    constructor() {
        insurer = msg.sender;
    }

    // Customer files claim
    function fileClaim(
        uint _amount,
        string memory _description
    ) external {
        require(
            _amount > 0,
            "Claim amount must be greater than zero"
        );

        claimCounter++;

        claims[claimCounter] = Claim(
            claimCounter,
            payable(msg.sender),
            _amount,
            ClaimStatus.Filed,
            _description
        );

        emit ClaimFiled(
            claimCounter,
            msg.sender,
            _amount,
            _description
        );
    }

    // Insurer approves claim
    function approveClaim(
        uint _id
    ) external onlyInsurer {
        require(
            _id > 0 && _id <= claimCounter,
            "Invalid claim ID"
        );

        require(
            claims[_id].status == ClaimStatus.Filed,
            "Claim not filed or already processed"
        );

        claims[_id].status = ClaimStatus.Approved;

        emit ClaimApproved(_id);
    }

    // Insurer rejects claim
    function rejectClaim(
        uint _id
    ) external onlyInsurer {
        require(
            _id > 0 && _id <= claimCounter,
            "Invalid claim ID"
        );

        require(
            claims[_id].status == ClaimStatus.Filed,
            "Claim not valid for rejection"
        );

        claims[_id].status = ClaimStatus.Rejected;

        emit ClaimRejected(_id);
    }

    // Insurer pays approved claim
    function payClaim(
        uint _id
    ) external onlyInsurer {
        require(
            _id > 0 && _id <= claimCounter,
            "Invalid claim ID"
        );

        Claim storage claim = claims[_id];

        require(
            claim.status == ClaimStatus.Approved,
            "Claim not approved"
        );

        require(
            address(this).balance >= claim.amount,
            "Insufficient contract balance"
        );

        claim.status = ClaimStatus.Paid;

        (bool success, ) = claim.customer.call{
            value: claim.amount
        }("");

        require(
            success,
            "Payment transfer failed"
        );

        emit ClaimPaid(
            _id,
            claim.amount
        );
    }

    // Fund contract
    function fundContract()
        external
        payable
        onlyInsurer
    {
        require(
            msg.value > 0,
            "Send ETH to fund contract"
        );
    }

    // Get only claim status
    function getClaimStatus(
        uint _id
    ) external view returns (ClaimStatus) {
        require(
            _id > 0 && _id <= claimCounter,
            "Invalid claim ID"
        );

        return claims[_id].status;
    }

    // View single claim details (without exposing unnecessary UI data)
    function getClaim(
        uint _id
    )
        external
        view
        returns (
            uint id,
            uint amount,
            ClaimStatus status,
            string memory description
        )
    {
        require(
            _id > 0 && _id <= claimCounter,
            "Invalid claim ID"
        );

        Claim memory claim = claims[_id];

        return (
            claim.id,
            claim.amount,
            claim.status,
            claim.description
        );
    }

    // NEW: Get all claims for scrollable frontend list
    function getAllClaims()
        external
        view
        returns (
            uint[] memory ids,
            uint[] memory amounts,
            string[] memory descriptions
        )
    {
        ids = new uint[](claimCounter);
        amounts = new uint[](claimCounter);
        descriptions = new string[](claimCounter);

        for (uint i = 1; i <= claimCounter; i++) {
            Claim memory claim = claims[i];

            ids[i - 1] = claim.id;
            amounts[i - 1] = claim.amount;
            descriptions[i - 1] = claim.description;
        }

        return (
            ids,
            amounts,
            descriptions
        );
    }

    receive() external payable {}
}
